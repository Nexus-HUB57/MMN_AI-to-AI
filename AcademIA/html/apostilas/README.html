🔧 Como Compilar para PDF

# Requer pandoc + wkhtmltopdf
cd AcademIA/apostilas/
for f in *.md; do
  pandoc "$f" -o "../pdf/$(basename "$f" .md).pdf" \
    --pdf-engine=wkhtmltopdf \
    --css=../templates/pdf.css \
    --toc --toc-depth=2
done
